public class App {

        double resto = num1 % num2;

                break;

            case 3:
                System.out.println("Tu bebida es Gaseosa");

            case 4:
                System.out.println("Tu bebida es Vino");
                break;
            case 5:

                // FOR
        for (int i = 0; i <= 3; i++) {

// Estructuras de datos
